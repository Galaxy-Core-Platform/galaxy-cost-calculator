üíª Quick Start

# Clone the repository git clone https://github.com/Galaxy-Core-Platform/orion.git cd orion # Set up environment variables export DATABASE_URL="postgresql://user:pass@localhost/orion" export REDIS_URL="redis://localhost:6379" # Build and run cargo build --release cargo run # Or use Docker docker build -t orion:latest . docker run -p 5020:5020 orion:latest # Health check curl http://localhost:5020/health
